function parseTargetTimeSecs(timeStr) {
    // mm:ss.d
    const parts = timeStr.split(':');
    if (parts.length !== 2) return 0;
    const m = parseInt(parts[0]) || 0;
    const s = parseFloat(parts[1]) || 0;
    return m * 60 + s;
}

function formatTime(totalSecs) {
    let m = Math.floor(totalSecs / 60);
    let s = (totalSecs % 60).toFixed(1);
    return `${m.toString().padStart(2, '0')}:${s.padStart(4, '0')}`;
}

function setupRaceStrategy() {
    // Pace base richiesto = Energia per percorrere distanza nel tempo dato
    // E = (2.8/Pace)^3 approx Concept2 formula: Pace(s/500m) = 500 * (2.8 / (Watts)^(1/3))
    // Watts = (2.8 / (PaceSec/500))^3
    const pace500 = (raceTargetTimeSec / raceDistanceTotal) * 500;
    raceBasePaceW = Math.pow(2.8 / (pace500), 3) * Math.pow(500, 3); // ??? Wait, Formula Concept2 W = 2.8 / (pace in sec/meter)^3
    // v = 500 / pace500 (m/s)
    // W = 2.8 * v^3
    const v = raceDistanceTotal / raceTargetTimeSec;
    raceBasePaceW = 2.8 * Math.pow(v, 3);
    
    // Default SPM basato sulla distanza
    if (raceDistanceTotal <= 500) raceTargetSpm = 36;
    else if (raceDistanceTotal <= 1000) raceTargetSpm = 32;
    else if (raceDistanceTotal <= 2000) raceTargetSpm = 28;
    else raceTargetSpm = 24;
}

            startActualRace();
            setTimeout(() => {
                if (currentRaceState === 'RACING') {
                    document.getElementById('race-countdown-overlay').style.display = 'none';
                }
            }, 1000);
            return;
        }

        if (raceCountdownSec > 0) {
            valObj.innerText = raceCountdownSec;
            if (raceCountdownSec <= 3) valObj.style.color = '#ef4444';
        } else {
            // GO
            valObj.innerText = "GO!";
            valObj.style.color = '#10b981';
            currentRaceState = 'RACING';
            startActualRace();
            setTimeout(() => {
                if (currentRaceState === 'RACING') {
                    document.getElementById('race-countdown-overlay').style.display = 'none';
                }
            }, 1000);
        }
    } else if (currentRaceState === 'RACING') {
        raceElapsedSec++;
        wState.elapsedSec = raceElapsedSec; // Sincronizza timer per CSV
        
        // Aggiorna target in base alla distanza corrente
        updateRaceTargets();
        
        // Controlla fine
        let currentDist = telemetry.distance - raceStartDistOffset;
        if (currentDist >= raceDistanceTotal) {
            finishRace();
        } else {
            // Aggiorna progress bar gara
            let p = (currentDist / raceDistanceTotal) * 100;
            if (p < 0) p = 0;
            if (p > 100) p = 100;
            document.getElementById('wp-cursor').style.left = `${p}%`;
            let fill = document.getElementById('race-progress-fill');
            if (fill) fill.style.width = `${p}%`;
            let rem = raceDistanceTotal - currentDist;
            if (rem < 0) rem = 0;
            document.getElementById('wp-remaining').innerText = `${Math.floor(rem)}m rem`;
            let eM = Math.floor(raceElapsedSec / 60); let eS = raceElapsedSec % 60;
            document.getElementById('wp-elapsed').innerText = `${eM.toString().padStart(2,'0')}:${eS.toString().padStart(2,'0')}`;
        }
    }
}

function startActualRace() {
    raceStartDistOffset = telemetry.distance;
    raceElapsedSec = 0;
    
    // Aggiorna target
    wState.targetWatts = raceBasePaceW;
    wState.targetSpm = raceTargetSpm;
    wState.targetForce = wState.targetWatts > 0 ? (wState.targetWatts / 4.0) : 0;
    
    // Aggiorna titolo
    document.getElementById('det-title').innerText = "Gara " + raceDistanceTotal + "m";
}

function finishRace() {
    clearInterval(raceTimerInterval);
    currentRaceState = 'FINISHED';

    let pbKey = 'pb_' + pendingRaceDistance;
    let oldPb = localStorage.getItem(pbKey);
    if (!oldPb || raceElapsedSec < parseInt(oldPb)) {
        localStorage.setItem(pbKey, raceElapsedSec);
        if (ws && ws.readyState === WebSocket.OPEN) {
            ws.send(`SAVE_PB:${pendingRaceDistance},${raceElapsedSec}`);
        }
    }

    document.getElementById('races-grid').style.display = 'grid';
    document.getElementById('race-active-panel').style.display = 'none';
    finishWorkout(); // Stop logging and show summary
    alert("GARA COMPLETATA! Tempo: " + formatTime(raceElapsedSec));
    renderRacesGrid(); // Aggiorna UI per mostrare eventuale nuovo PB
}

function updateRaceTargets() {
    let currentDist = telemetry.distance - raceStartDistOffset;
    if (currentDist < 0) currentDist = 0;
    
    let wMultiplier = 1.0;
    let spmOffset = 0;
    
    if (raceDistanceTotal <= 500) {
        // Sprint All out
        wMultiplier = 1.1; 
        spmOffset = +2;
    } else if (raceDistanceTotal === 1000) {
        if (currentDist < 150) { wMultiplier = 1.05; spmOffset = +2; }
        else if (currentDist > 750) { wMultiplier = 1.08; spmOffset = +4; }
    } else if (raceDistanceTotal === 2000) {
        if (currentDist < 250) { wMultiplier = 1.03; spmOffset = +2; }
        else if (currentDist > 1500) { wMultiplier = 1.05; spmOffset = +2; }
        else if (currentDist > 1750) { wMultiplier = 1.10; spmOffset = +4; }
    } else if (raceDistanceTotal >= 5000) {
        if (currentDist < 500) { wMultiplier = 0.95; } // start controllato
        else if (currentDist > raceDistanceTotal - 500) { wMultiplier = 1.05; spmOffset = +2; }
    }
    
    wState.targetWatts = raceBasePaceW * wMultiplier;
    wState.targetSpm = raceTargetSpm + spmOffset;
    wState.targetForce = wState.targetWatts > 0 ? (wState.targetWatts / 4.0) : 0;
}

function connectWS(ip) {
    wsLastIp = ip;
    if (wsReconnectTimer) { clearTimeout(wsReconnectTimer); wsReconnectTimer = null; }
    // Chiudi il vecchio socket azzerando gli handler, cosÃ¬ il suo onclose non
    // scatena una seconda riconnessione sovrapposta.
    if (ws) {
        ws.onopen = ws.onclose = ws.onerror = ws.onmessage = null;
        try { ws.close(); } catch(e) {}
    }
    let url = ip ? `ws://${ip}/ws` : `ws://${window.location.hostname}/ws`;
    
    const ind = document.getElementById('ws-status');
    const dot = ind.querySelector('.dot');
    const txt = ind.querySelector('.text');
    
    txt.innerText = 'Connessione...';
    dot.className = 'dot';
    
    try {
        ws = new WebSocket(url);
        
        ws.onopen = () => {
            wsReconnectDelay = 500;          // riconnessione riuscita: resetta il backoff
            window.lastSyncTime = Date.now(); // concedi 2s di grazia al watchdog sulla nuova connessione
            txt.innerText = 'Connesso';
            dot.className = 'dot connected';
            ws.send('GET_CFG');              // niente SCAN automatico: la scansione resta manuale (pulsante UI)
            ws.send('GET_PBS');              // recupera PB gare
        };
        
        ws.onclose = () => {
            txt.innerText = 'Disconnesso';
            dot.className = 'dot disconnected';
            scheduleReconnect();
        };
        
        ws.onerror = () => {
            txt.innerText = 'Errore WS';
            dot.className = 'dot disconnected';
            // Nessun reconnect qui: onerror Ã¨ seguito da onclose, che lo gestisce.
        };
        
        ws.binaryType = "arraybuffer";
        ws.onmessage = (msg) => {
            if (msg.data instanceof ArrayBuffer) {
                window.lastSyncTime = Date.now();
                let floats = new Float32Array(msg.data);
                
                for(let i=0; i<floats.length; i+=3) {
                    let fKg = floats[i];
                    let cord = floats[i+1];
                    let seat = floats[i+2];
                    
                    if (isNaN(fKg)) break;
                    if (isNaN(seat)) seat = 0;
                    
                    telemetry.forceLive = fKg;
                    telemetry.cordPos = cord;
                    
                    document.getElementById('val-force').innerText = fKg.toFixed(1);
                    document.getElementById('val-cord').innerText = cord.toFixed(2);
                    document.getElementById('stat-force-live').innerHTML = `${fKg.toFixed(1)} <small>kg</small>`;
                    
                    forceHistory.push(fKg);
                    if (forceHistory.length > maxHistorySamples) forceHistory.shift();
                    cordHistory.push(cord);
                    if (cordHistory.length > maxHistorySamples) cordHistory.shift();
                    seatHistory.push(seat);
                    if (seatHistory.length > maxHistorySamples) seatHistory.shift();
                    
                    if (fKg > telemetry.peakForce) telemetry.peakForce = fKg;
                    
                    if (wState.running && wState.startTime > 0) {
                        if (wState.sampleCount === undefined) wState.sampleCount = 0;
                        
                        accForce += fKg;
                        accCord += cord;
                        accSeat += seat;
                        accCount++;
                        wState.sampleCount++;
                        if (accCount >= 1) { // Nessuna media, piena frequenza reale
                            workoutRawData.push({
                                time: Date.now() - window.recordStartMs,
                                tw: wState.targetWatts || 0,
                                ts: wState.targetSpm || 0,
                                tf: wState.targetForce || 0,
                                rw: telemetry.watts || 0,
                                rf: accForce / accCount || 0,
                                cp: accCord / accCount || 0,
                                sp: accSeat / accCount || 0,
                                hr: telemetry.heartRate || 0
                            });
                            accForce = 0; accCord = 0; accSeat = 0; accCount = 0;
                        }
                    }
                }
                return;
            }

            let txt = msg.data.trim();
            if (txt.startsWith('CFG:')) {
                const allparts = txt.substring(4).split('|');
                const parts = allparts[0].split(',');
                if(parts.length >= 11) {
                    window.currentTara = parts[0];
                    window.currentScala = parts[1];
                    const elTara = document.getElementById('cfg-tara'); if(elTara) elTara.value = parts[0];
                    const elScala = document.getElementById('cfg-scala'); if(elScala) elScala.value = parts[1];
                    const elHeight = document.getElementById('u-height'); if(elHeight) elHeight.value = parts[2];
                    const elWeight = document.getElementById('u-weight'); if(elWeight) elWeight.value = parts[3];
                    const elPull = document.getElementById('cfg-pull'); if(elPull) elPull.value = parts[4];
                    const elRel = document.getElementById('cfg-rel'); if(elRel) elRel.value = parts[5];
                    const elPpr = document.getElementById('cfg-encppr'); if(elPpr) elPpr.value = parts[6];
                    const elCirc = document.getElementById('cfg-pullcirc'); if(elCirc) elCirc.value = parts[7];
                    const elLoff = document.getElementById('cfg-laseroffset'); if(elLoff) elLoff.value = parts[8];
                    
                    const elSsid = document.getElementById('cfg-wifissid');
                    if(elSsid) {
                        let opt = document.createElement('option');
                        opt.value = parts[9];
                        opt.text = parts[9] === "" ? "Nessuna (AP)" : parts[9];
                        elSsid.appendChild(opt);
                        elSsid.value = parts[9];
                    }
                    const elPass = document.getElementById('cfg-wifipass'); if(elPass) elPass.value = parts[10];
                    
                    if (parts[9] === "RP_Handle") {
                        window.isStandaloneHandle = true;
                        // Nascondi elementi Telaio-only
                        document.querySelectorAll('.stat-telaio-only').forEach(el => el.style.display = 'none');
                    }
                    
                    if(parts.length >= 12) {
                        const elFtp = document.getElementById('u-ftp'); if(elFtp) elFtp.value = parts[11];
                    }
                }
            } else if (txt.startsWith('PBS:')) {
                const pairs = txt.substring(4).split(',');
                pairs.forEach(p => {
                    let [d, t] = p.split('=');
                    if (d && t && parseInt(t) > 0) {
                        localStorage.setItem('pb_' + d, t);
                    }
                });
                renderRacesGrid();
            } else if (txt.startsWith('WIFI_LIST:')) {
                const ssidList = txt.substring(10).split(',');
                const sel = document.getElementById('cfg-wifissid');
                if (sel) {
                    const currentVal = sel.value;
                    sel.innerHTML = '<option value="">Nessuna (Usa ModalitÃ¯Â¿Â½ AP)</option>';
                    let currentValRestored = false;
                    ssidList.forEach(ssid => {
                        if (ssid.trim() !== '') {

                            option.value = ssid; option.text = ssid;
                            sel.appendChild(option);
                            if(ssid === currentVal) currentValRestored = true;
                        }
                    });
                    if(!currentValRestored && currentVal !== "") {
                        const option = document.createElement('option');
                        option.value = currentVal; option.text = currentVal + " (Fuori portata)";
                        sel.appendChild(option);
                    }
                    sel.value = currentVal;
                }
            } else if (txt.startsWith('METRICS:')) {
                let p = txt.substring(8).split('|');
                if(p.length > 12) {
                    telemetry.watts = parseInt(p[0]);
                    telemetry.wattsLive = telemetry.watts;
                    telemetry.distance = parseInt(p[1]);
                    telemetry.spm = parseInt(p[2]);

                    telemetry.paceSecs = parseInt(p[3]);

                    let elFkg = document.getElementById('diag-fkg'); if(elFkg) elFkg.innerText = parseFloat(p[11]).toFixed(2) + ' kg';
                    let elPull = document.getElementById('diag-pull'); if(elPull) elPull.innerText = p[12];
                    if (p.length > 13) {
                        let elCord = document.getElementById('diag-cord'); if(elCord) elCord.innerText = parseFloat(p[13]).toFixed(3) + ' m';

                    document.getElementById('stat-dist').innerHTML = `${dispDist} <small>m</small>`;
                    document.getElementById('stat-kcal').innerHTML = `${telemetry.kcal.toFixed(1)} <small>kcal</small>`;
                    
                    if (telemetry.paceSecs > 0) {
                        let pm = Math.floor(telemetry.paceSecs / 60);
                        let ps = Math.floor(telemetry.paceSecs % 60);
                        document.getElementById('val-pace').innerText = `${pm}:${ps.toString().padStart(2, '0')}`;
                    } else {
                        document.getElementById('val-pace').innerText = '--:--';
                    }
                }
            }
        };
    } catch(e) {
        txt.innerText = 'Url non valido';
        dot.className = 'dot disconnected';
    }

        }
    }
}

// --- WORKOUT LIBRARY & ENGINE ---
let allWorkouts = [];
let activeTags = new Set();
let currentFilter = { text: '', category: 'all', durationMin: 0, durationMax: 120 };

function getPhaseColor(pwr) {
    if (pwr >= 1.05) return '#ef4444'; // Red (Sprint/Anaerobic)
    if (pwr >= 0.9) return '#f59e0b'; // Yellow (Threshold)
    if (pwr >= 0.75) return '#10b981'; // Green (Tempo)
    if (pwr > 0) return '#3b82f6'; // Blue (Endurance)
    return '#94a3b8'; // Grey (Recovery)
}

function initWorkoutLibrary() {
    if (!window.WORKOUTS_DATA) {
        document.getElementById('workouts-grid').innerHTML = '<p>Nessun allenamento trovato (workouts.js non caricato).</p>';
        return;
    }
    
    // Filtriamo i Time Trials e zzConcept2 Verified
    allWorkouts = window.WORKOUTS_DATA.filter(w => {
        let isTimeTrialCat = w.category && w.category.toLowerCase().includes('time trial');
        let isTimeTrialTag = w.tags && w.tags.some(t => t.toLowerCase().includes('time trial'));
        let isC2Cat = w.category && w.category.toLowerCase().includes('zzconcept2');
        return !isTimeTrialCat && !isTimeTrialTag && !isC2Cat;
    });
    
    // Populate Categories
    let cats = [...new Set(allWorkouts.map(w => w.category))].sort();
    const catSel = document.getElementById('filter-category');
    cats.forEach(c => {
        let opt = document.createElement('option');
        opt.value = c; opt.innerText = c;
        catSel.appendChild(opt);
    });
    
    // Populate Tags
    let tags = new Set();
    allWorkouts.forEach(w => w.tags && w.tags.forEach(t => tags.add(t)));
    const tagCont = document.getElementById('filter-tags');
    [...tags].sort().forEach(t => {
        let span = document.createElement('span');
        span.className = 'tag';
        span.innerText = t;
        span.addEventListener('click', () => {
            if (activeTags.has(t)) { activeTags.delete(t); span.classList.remove('active'); }
            else { activeTags.add(t); span.classList.add('active'); }
            applyFilters();
        });
        tagCont.appendChild(span);
    });
    
    // Event Listeners
    catSel.addEventListener('change', e => {
        currentFilter.category = e.target.value;
        applyFilters();
    });
    document.getElementById('filter-duration').addEventListener('input', e => {
        currentFilter.durationMax = parseInt(e.target.value);
        document.getElementById('filter-duration-val').innerText = currentFilter.durationMax;
        
        // Prevent max from being less than min
        if (currentFilter.durationMax < currentFilter.durationMin) {
            currentFilter.durationMin = currentFilter.durationMax;
            document.getElementById('filter-duration-min').value = currentFilter.durationMin;
            document.getElementById('filter-duration-min-val').innerText = currentFilter.durationMin;
        }
        applyFilters();
    });

    document.getElementById('filter-duration-min').addEventListener('input', e => {
        currentFilter.durationMin = parseInt(e.target.value);
        document.getElementById('filter-duration-min-val').innerText = currentFilter.durationMin;
        
        // Prevent min from being greater than max
        if (currentFilter.durationMin > currentFilter.durationMax) {
            currentFilter.durationMax = currentFilter.durationMin;
            document.getElementById('filter-duration').value = currentFilter.durationMax;
            document.getElementById('filter-duration-val').innerText = currentFilter.durationMax;
        }
        applyFilters();
    });
    
    applyFilters(); // Initial render
}

function applyFilters() {
    let filtered = allWorkouts.filter(w => {
        let durMin = Math.round(w.duration / 60);
        if (durMin < currentFilter.durationMin) return false;
        if (currentFilter.durationMax < 120 && durMin > currentFilter.durationMax) return false;
        if (currentFilter.category !== 'all' && w.category !== currentFilter.category) return false;
        if (activeTags.size > 0) {
            if (!w.tags) return false;
            let hasTag = false;
            for(let t of activeTags) { if (w.tags.includes(t)) hasTag = true; }
            if (!hasTag) return false;
        }
        return true;
    });
    
    // Sort alphabetically by name (letters first, numbers at the end)
    filtered.sort((a, b) => {
        let isNumA = /^[0-9]/.test(a.name);
        let isNumB = /^[0-9]/.test(b.name);
        if (isNumA && !isNumB) return 1;
        if (!isNumA && isNumB) return -1;
        return a.name.localeCompare(b.name);
    });
    
    document.getElementById('workout-count').innerText = `${filtered.length} risultati`;
    const grid = document.getElementById('workouts-grid');
    grid.innerHTML = '';
    
    filtered.forEach(w => {
        let card = document.createElement('div');
        card.className = 'workout-card';
        card.innerHTML = `
            <div class="workout-cat">${w.category}</div>
            <div class="workout-title">${w.name}</div>
            <div class="workout-profile"></div>
            <div style="font-size:0.8rem; color:var(--text-dim); margin-bottom:10px;">${w.description ? w.description.substring(0,80)+'...' : ''}</div>
            <div class="workout-meta">
                <span>${Math.round(w.duration / 60)} min</span>
                <span>${w.tags ? w.tags.join(', ') : ''}</span>
            </div>
        `;
        
        // Disegna il profilo grafico (l'immagine dell'allenamento)
        let profContainer = card.querySelector('.workout-profile');
        let steps = parseZwoSteps(w);
        let maxPwr = 0;
        steps.forEach(s => { if(s.pwr > maxPwr) maxPwr = s.pwr; });
        if (maxPwr === 0) maxPwr = 1.2;
        
        steps.forEach(s => {
            let widthPct = (s.dur / w.duration) * 100;
            let heightPct = (s.pwr / maxPwr) * 100;
            if(heightPct < 5) heightPct = 5;
            
            let color = s.color || getPhaseColor(s.pwr);
            
            let bar = document.createElement('div');
            bar.style.width = widthPct + '%';
            bar.style.height = heightPct + '%';
            bar.style.backgroundColor = color;
            profContainer.appendChild(bar);
        });

        card.addEventListener('click', () => showWorkoutDetails(w));
        grid.appendChild(card);
    });
}

function showWorkoutDetails(w) {
    activeWorkout = w;

function parseZwoSteps(w) {
    let steps = [];
    if (!w.steps) return steps;
    w.steps.forEach(st => {
        let dur = parseInt(st.Duration || 0);
        let pwr = parseFloat(st.Power || st.PowerLow || 0);
        let cad = parseInt(st.Cadence || 24); // Valore di default se non specificato
        
        if (st.type === 'SteadyState' || st.type === 'Warmup' || st.type === 'Cooldown') {
            steps.push({ name: st.type, dur, pwr, cad });
        } else if (st.type === 'IntervalsT') {
            let repeats = parseInt(st.Repeat || 1);
            let onDur = parseInt(st.OnDuration || 0);
            let offDur = parseInt(st.OffDuration || 0);
            let onPwr = parseFloat(st.OnPower || 0);
            let offPwr = parseFloat(st.OffPower || 0);
            for (let i=0; i<repeats; i++) {
                steps.push({ name: `Sprint ${i+1}/${repeats}`, dur: onDur, pwr: onPwr, cad: cad, color: 'var(--color-work)' });
                steps.push({ name: `Recupero ${i+1}/${repeats}`, dur: offDur, pwr: offPwr, cad: 0, color: 'var(--color-rest)' });
            }
        }
    });
    return steps;
}

function startWorkout(w) {
    let parsedSteps = parseZwoSteps(w);
    if(parsedSteps.length === 0) { alert('Formato allenamento non supportato'); return; }
    
    // Attiva NoSleep per impedire spegnimento schermo
    if (typeof noSleep !== 'undefined') noSleep.enable().catch(function(e){ console.warn('NoSleep init error:', e); });
    
    // Switch to dashboard
    document.querySelector('.nav-btn[data-target="tab-dashboard"]').click();
    
    wState = {
        running: true,
        steps: parsedSteps,
        totalSec: w.duration,
        elapsedSec: 0,
        stepIndex: 0,
        stepTimeLeft: parsedSteps[0].dur,
        startTime: Date.now()
    };
    
    window.recordStartMs = wState.startTime;
    wState.sampleCount = 0;
    
    document.getElementById('workout-progress-container').style.display = 'block';
    document.getElementById('btn-stop-wo').style.display = 'block';
    workoutRawData = []; // Reset raw data
    
    // Build track HTML
    let trackHtml = '';
    parsedSteps.forEach(s => {
        let pct = (s.dur / wState.totalSec) * 100;
        let c = s.color || getPhaseColor(s.pwr);
        trackHtml += `<div style="width:${pct}%; background:${c}; height:100%; border-right:1px solid rgba(0,0,0,0.5);"></div>`;
    });
    document.getElementById('wp-track').innerHTML = trackHtml;
    
    if (workoutInterval) clearInterval(workoutInterval);
    workoutInterval = setInterval(workoutTick, 1000);
    workoutTick();
}

function workoutTick() {
    if (!wState.running) return;
    
    if (wState.stepTimeLeft <= 0) {
        wState.stepIndex++;
        if (wState.stepIndex >= wState.steps.length) {
            finishWorkout();
            return;
        }
        wState.stepTimeLeft = wState.steps[wState.stepIndex].dur;
    }
    
    let step = wState.steps[wState.stepIndex];
    // ZWO Power is fraction of FTP
    wState.targetWatts = step.pwr > 0 ? Math.round(step.pwr * config.ftp) : 0;
    wState.targetSpm = step.cad || 0;
    
    // Update Target Labels
    document.getElementById('tgt-spm').innerText = wState.targetSpm > 0 ? `Target: ${wState.targetSpm}` : '';
    document.getElementById('tgt-force').innerText = wState.targetWatts > 0 ? `Target: ${wState.targetWatts}W` : '';
    // Approx mapping Watts to peak force (just for visualization)
    wState.targetForce = wState.targetWatts > 0 ? (wState.targetWatts / 4.0) : 0;
    
    // Dynamic color for power metric
    let phaseColor = step.color || getPhaseColor(step.pwr);
    document.getElementById('val-watts').style.color = phaseColor;
    
    // Store for CSV even if WebSocket isn't updating fast enough (guarantees data exists)
    workoutRawData.push({
        time: Date.now() - window.recordStartMs,
        tw: wState.targetWatts,
        ts: wState.targetSpm,
        tf: wState.targetForce,
        rw: telemetry.watts,
        rf: telemetry.forceLive,
        cp: telemetry.cordPos,
        sp: 0,
        hr: telemetry.heartRate || 0
    });
    
    // Update Progress UI
    let p = (wState.elapsedSec / wState.totalSec) * 100;
    document.getElementById('wp-cursor').style.left = `${p}%`;
    
    let eM = Math.floor(wState.elapsedSec / 60); let eS = wState.elapsedSec % 60;
    document.getElementById('wp-elapsed').innerText = `${eM.toString().padStart(2,'0')}:${eS.toString().padStart(2,'0')}`;
    
    let rem = wState.totalSec - wState.elapsedSec;
    let rM = Math.floor(rem / 60); let rS = rem % 60;
    document.getElementById('wp-remaining').innerText = `-${rM.toString().padStart(2,'0')}:${rS.toString().padStart(2,'0')}`;
    
    let stM = Math.floor(wState.stepTimeLeft / 60); let stS = wState.stepTimeLeft % 60;
    document.getElementById('wp-next').innerText = `${step.name} (${stM.toString().padStart(2,'0')}:${stS.toString().padStart(2,'0')})`;
    
    wState.elapsedSec++;
    wState.stepTimeLeft--;
}

function finishWorkout() {
    // Disattiva NoSleep
    if (typeof noSleep !== 'undefined') noSleep.disable();
    wState.running = false;
    clearInterval(workoutInterval);
    document.getElementById('workout-progress-container').style.display = 'none';
    document.getElementById('btn-stop-wo').style.display = 'none';
    document.getElementById('tgt-spm').innerText = '';
    document.getElementById('tgt-force').innerText = '';
    document.getElementById('val-watts').style.color = ''; // Reset color
    
    // Avvia l'esportazione del CSV a 1kHz

    document.getElementById('wo-dist').innerText = document.getElementById('stat-dist').innerText;
    document.getElementById('wo-strokes').innerText = telemetry.strokes;
    document.getElementById('wo-kcal').innerText = document.getElementById('stat-kcal').innerText;
    
    document.getElementById('workout-overlay').classList.add('active');
}

function exportCSV() {
    if (workoutRawData.length < 2) return;
    
    let btn = document.getElementById('btn-stop-wo');
    if(btn) btn.style.display = 'block';
    
    // Elaborazione asincrona per non bloccare la UI
    setTimeout(() => {
        let maxTime = workoutRawData[workoutRawData.length - 1].time;
        let csvArr = [];
        
        // Metadati Allenamento all'inizio del file
        if (activeWorkout) {
            csvArr.push(`"${(activeWorkout.name || 'Custom Workout').replace(/"/g, '""')}"`);
            csvArr.push(`"${(activeWorkout.description || '').replace(/"/g, '""')}"`);
            csvArr.push("Durata Prevista");
            csvArr.push(`${Math.round(activeWorkout.duration / 60)} min`);
            csvArr.push("TSS (Stress Score) stimato");
            csvArr.push(`${activeWorkout.tss || 0}`);
            csvArr.push(""); // riga vuota
        }
        
        csvArr.push('time_ms,target_watts,target_spm,target_force,real_watts,real_force,cord_pos,seat_pos,heart_rate');
        
        let dIdx = 0;
        let d0 = workoutRawData[0];
        let d1 = workoutRawData[1];
        
        for (let t = 0; t <= maxTime; t += 10) { // Step 10ms (100Hz)
            while (t > d1.time && dIdx < workoutRawData.length - 2) {
                dIdx++;
                d0 = workoutRawData[dIdx];
                d1 = workoutRawData[dIdx + 1];
            }
            
            // Interpolazione lineare
            let pct = 0;
            if (d1.time > d0.time) pct = (t - d0.time) / (d1.time - d0.time);
            if (pct < 0) pct = 0; if (pct > 1) pct = 1;
            
            let tw = (d0.tw + (d1.tw - d0.tw) * pct).toFixed(1);
            let ts = (d0.ts + (d1.ts - d0.ts) * pct).toFixed(1);
            let tf = (d0.tf + (d1.tf - d0.tf) * pct).toFixed(2);
            let rw = (d0.rw + (d1.rw - d0.rw) * pct).toFixed(1);
            let rf = (d0.rf + (d1.rf - d0.rf) * pct).toFixed(2);
            let cp = (d0.cp + (d1.cp - d0.cp) * pct).toFixed(3);
            let sp = (d0.sp + (d1.sp - d0.sp) * pct).toFixed(3);
            let hr = Math.round(d0.hr + (d1.hr - d0.hr) * pct);
            csvArr.push(`${t},${tw},${ts},${tf},${rw},${rf},${cp},${sp},${hr}`);
        }
        
        let csvContent = csvArr.join('\n');
        let blob = new Blob([csvContent], { type: 'text/csv;charset=utf-8;' });
        let url = URL.createObjectURL(blob);
        let a = document.createElement("a");
        let dateStr = new Date().toISOString().replace(/[:.]/g, '-');
        a.href = url;
        a.download = `SmartRower_Export_${dateStr}.csv`;
        document.body.appendChild(a);
        a.click();
        document.body.removeChild(a);
        
        if(btn) btn.style.display = 'none';
    }, 100);
}

function renderGraph() {
    canvas.width = canvas.clientWidth;
    canvas.height = canvas.clientHeight;
    ctx.clearRect(0,0, canvas.width, canvas.height);
    
    let w = canvas.width;
    let h = canvas.height;
    // Lascia spazio in basso per la progress bar
    let chartH = h - 40; 
    
    // Calcolo Y dinamico
    let maxF = 40; // Base/minimo max kg per forza
    if (wState.running && wState.targetForce > 0) {
        maxF = Math.max(maxF, wState.targetForce * 1.3); // Lascia un 30% di headroom sopra il target
    }
    for(let f of forceHistory) {
        if(f > maxF * 0.9) maxF = f * 1.2; // Scala dinamicamente se l'utente supera il target o i 40kg
    }
    
    let maxD = 1.6; // Default max metri per corda/sellino
    
    // Draw Ghost Curves se siamo in allenamento
    if (wState.running && wState.targetSpm > 0 && wState.targetForce > 0) {
        let t = Date.now() / 1000.0;
        let T = 60.0 / wState.targetSpm;
        
        // Calcolo fisico del Duty Cycle (Rapporto Trazione/Recupero)
        // Nel canottaggio reale, il tempo di trazione (drive) ÃƒÂ¨ relativamente costante (circa 0.8s - 0.9s).
        // ÃƒË† il tempo di recupero che varia per cambiare i colpi al minuto (SPM).
        let baseDriveTime = 0.95 - (Math.max(0, wState.targetSpm - 20) * 0.01); 
        let duty = baseDriveTime / T;
        // Limitiamo il duty cycle in limiti realistici (tra il 25% e il 55%)
        duty = Math.max(0.25, Math.min(0.55, duty));
        
        let targetStrokeLength = (config.height * 0.75) / 100.0;
        let seatStrokeLength = targetStrokeLength * 0.65;
        
        let forcePts = [];
        let cordPts = [];
        let seatPts = [];
        
        for (let i=0; i<maxHistoryMs; i++) {
            let offset = (maxHistoryMs - 1 - i) * 0.001; // 1ms resolution
            let tpt = t - offset;
            let phase = (tpt % T) / T;
            
            let forceY = 0;
            let cordY = 0;
            let seatY = 0;
            
            if (phase <= duty) {
                // FASE DI TRAZIONE (Drive)
                let p = phase / duty; // 0 to 1
                
                // Forza: calcolo dinamico basato sul Super Coach (Valery Kleshnev model)
                let mode = document.getElementById('cfg-coach') ? document.getElementById('cfg-coach').value : 'auto';
                if(mode === 'auto') mode = wState.targetSpm > 26 ? 'race' : 'endurance';
                
                if (mode === 'race') {
                    // Molto Front-loaded (Picco al ~25%)
                    forceY = wState.targetForce * Math.sin(Math.pow(p, 0.45) * Math.PI);
                } else {
                    // Endurance: piÃƒÂ¹ simmetrico, morbido (Picco al ~45%)
                    forceY = wState.targetForce * Math.sin(Math.pow(p, 0.85) * Math.PI);
                }
                
                // Corda: curva S
                cordY = targetStrokeLength * ((1 - Math.cos(p * Math.PI)) / 2);
                
                // Sellino: si muove prima della corda (finisce il suo corsa al 75% del drive)
                let pSeat = Math.min(1.0, p / 0.75);
                seatY = seatStrokeLength * ((1 - Math.cos(pSeat * Math.PI)) / 2);
                
            } else {
                // FASE DI RECUPERO (Recovery)
                let p = (phase - duty) / (1.0 - duty); // 0 to 1
                
                forceY = 0;
                
                // Corda: ritorno fluido
                cordY = targetStrokeLength * ((1 + Math.cos(p * Math.PI)) / 2);
                
                // Sellino: parte in ritardo (dopo aver disteso le braccia)
                let pSeat = Math.max(0, (p - 0.25) / 0.75);
                seatY = seatStrokeLength * ((1 + Math.cos(pSeat * Math.PI)) / 2);
            }
            
            let px = (i / (maxHistoryMs-1)) * w;
            forcePts.push({x: px, y: chartH - (forceY / maxF) * chartH});
            cordPts.push({x: px, y: chartH - (cordY / maxD) * chartH});
            seatPts.push({x: px, y: chartH - (seatY / maxD) * chartH});
        }
        
        // Disegna Ghost Force (Azzurro)
        ctx.beginPath();
        ctx.strokeStyle = 'rgba(34, 211, 238, 0.4)';
        ctx.lineWidth = 8;
        forcePts.forEach((pt, i) => i===0 ? ctx.moveTo(pt.x, pt.y) : ctx.lineTo(pt.x, pt.y));
        ctx.stroke();
        
        // Disegna Ghost Cord (Viola)
        ctx.beginPath();
        ctx.strokeStyle = 'rgba(168, 85, 247, 0.4)'; // color-dist
        ctx.setLineDash([5, 5]);
        ctx.lineWidth = 5;
        cordPts.forEach((pt, i) => i===0 ? ctx.moveTo(pt.x, pt.y) : ctx.lineTo(pt.x, pt.y));
        ctx.stroke();

        // Disegna Ghost Seat (Verde)
        ctx.beginPath();
        ctx.strokeStyle = 'rgba(16, 185, 129, 0.4)'; // color-rest
        ctx.lineWidth = 5;
        seatPts.forEach((pt, i) => i===0 ? ctx.moveTo(pt.x, pt.y) : ctx.lineTo(pt.x, pt.y));
        ctx.stroke();
        
        ctx.setLineDash([]); // Reset
    }
    
    // Draw Real History
    if (forceHistory.length > 1) {
        ctx.beginPath();
        ctx.strokeStyle = '#f59e0b';
        ctx.lineWidth = 3;
        
        for (let i=0; i<forceHistory.length; i++) {
            let dt = window.lastSyncTime ? (Date.now() - window.lastSyncTime) / 1000.0 : 0;
            if (dt > 0.13) dt = 0.13; // Clamp at 130ms
            let shiftIdxSamples = dt / (1.0 / SAMPLE_HZ);
            let idxOffset = (maxHistorySamples - forceHistory.length) + i;
            let px = ((idxOffset - shiftIdxSamples) / (maxHistorySamples-1)) * w;
            let py = chartH - (forceHistory[i] / maxF) * chartH;
            if (i===0) ctx.moveTo(px, py); else ctx.lineTo(px, py);
        }
        ctx.stroke();
    }
    
    // Draw Real History per Corda
    if (cordHistory.length > 1) {
        ctx.beginPath();
        ctx.strokeStyle = '#a855f7'; // Purple per Cavo
        ctx.lineWidth = 3;
        
        for (let i=0; i<cordHistory.length; i++) {
            let dt = window.lastSyncTime ? (Date.now() - window.lastSyncTime) / 1000.0 : 0;
            if (dt > 0.13) dt = 0.13;
            let shiftIdxSamples = dt / (1.0 / SAMPLE_HZ);
            let idxOffset = (maxHistorySamples - cordHistory.length) + i;
            let px = ((idxOffset - shiftIdxSamples) / (maxHistorySamples-1)) * w;
            let py = chartH - (cordHistory[i] / maxD) * chartH;
            if (i===0) ctx.moveTo(px, py); else ctx.lineTo(px, py);
        }
        ctx.stroke();
    }
    
    // Draw Real History per Sellino (Laser)
    if (seatHistory.length > 1) {
        ctx.beginPath();
        ctx.strokeStyle = '#10b981'; // Green per Sellino
        ctx.lineWidth = 3;
        
        for (let i=0; i<seatHistory.length; i++) {
            let dt = window.lastSyncTime ? (Date.now() - window.lastSyncTime) / 1000.0 : 0;
            if (dt > 0.13) dt = 0.13;
            let shiftIdxSamples = dt / (1.0 / SAMPLE_HZ);
            let idxOffset = (maxHistorySamples - seatHistory.length) + i;
            let px = ((idxOffset - shiftIdxSamples) / (maxHistorySamples-1)) * w;
            let py = chartH - (seatHistory[i] / maxD) * chartH;
            if (i===0) ctx.moveTo(px, py); else ctx.lineTo(px, py);
        }
        ctx.stroke();
    }
    
    requestAnimationFrame(renderGraph);
}
